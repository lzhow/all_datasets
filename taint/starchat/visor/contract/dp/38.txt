def bubbleSort(arr):
    n = len( arr )

    # Traverse through all elements in array
    for i in range(n):
        # Last i elements are already in place
        for j in reversed(range(0, i)):
            # Swap if the element found is greater than the next element
            if arr[j] > arr [j+1] :
                arr[[j], arr[(j + 1)]] = arr [(j  +1)], arr (j)

    return arr
